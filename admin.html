// --- ADD EXECUTIVE MEMBER --- document.getElementById('add-exec-btn').onclick = async () => { // This removes extra spaces or HTML tags if you accidentally paste them const rawUrl = document.getElementById('exec-url').value; const cleanUrl = rawUrl.includes('src="') ? rawUrl.split('src="')[1].split('"')[0] : rawUrl.trim(); try { await addDoc(collection(db, "executives"), { name: document.getElementById('exec-name').value.trim(), pos: document.getElementById('exec-pos').value.trim(), url: cleanUrl }); alert("Member Added Successfully!"); document.getElementById('exec-name').value = ""; document.getElementById('exec-pos').value = ""; document.getElementById('exec-url').value = ""; } catch (e) { alert("Error: " + e.message); } }; // --- UPDATE GALLERY --- document.getElementById('upload-gallery-btn').onclick = async () => { const rawUrls = document.getElementById('gallery-urls').value; // This splits by comma and cleans every single link in the list const urlArray = rawUrls.split(',').map(item => item.trim()).filter(item => item !== ""); try { await setDoc(doc(db, "website", "gallery"), { images: urlArray }); alert("Gallery Replaced with " + urlArray.length + " images!"); } catch (e) { alert("Error: " + e.message); } };
